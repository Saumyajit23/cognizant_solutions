import React, { useState } from 'react';

const CurrencyConvertor = () => {
  const [rupees, setRupees] = useState('');
  const [euros, setEuros] = useState('');

  const convertToEuro = () => {
    const converted = parseFloat(rupees) / 90;
    setEuros(converted.toFixed(2));
  };

  const convertToRupees = () => {
    const converted = parseFloat(euros) * 90;
    setRupees(converted.toFixed(2));
  };

  return (
    <div>
      <h2 style={{ color: 'green' }}>Currency Converter!!!</h2>

      <div>
        <label>Rupees: </label>
        <input
          type="number"
          value={rupees}
          onChange={(e) => setRupees(e.target.value)}
        />
        <button onClick={convertToEuro}>Convert to Euro</button>
      </div>

      <br />

      <div>
        <label>Euros: </label>
        <input
          type="number"
          value={euros}
          onChange={(e) => setEuros(e.target.value)}
        />
        <button onClick={convertToRupees}>Convert to Rupees</button>
      </div>
    </div>
  );
};

export default CurrencyConvertor;
